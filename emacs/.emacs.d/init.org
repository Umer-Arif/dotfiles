                                :ref nil :depth 1 :inherit ignore

                                :build (:not elpaca-activate)))

         (build (expand-file-name "elpaca/" elpaca-builds-directory))
         (order (cdr elpaca-order))
         (default-directory repo))
    (add-to-list 'load-path (if (file-exists-p build) build repo))
    (unless (file-exists-p repo)
      (make-directory repo t)
      (when (<= emacs-major-version 28) (require 'subr-x))
      (condition-case-unless-debug err
          (if-let* ((buffer (pop-to-buffer-same-window "*elpaca-bootstrap*"))
                    ((zerop (apply #'call-process `("git" nil ,buffer t "clone"
                                                    ,@(when-let* ((depth (plist-get order :depth)))
                                                        (list (format "--depth=%d" depth) "--no-single-branch"))
                                                    ,(plist-get order :repo) ,repo))))
                    ((zerop (call-process "git" nil buffer t "checkout"
                                          (or (plist-get order :ref) "--"))))
                    (emacs (concat invocation-directory invocation-name))
                    ((zerop (call-process emacs nil buffer nil "-Q" "-L" "." "--batch"
                                          "--eval" "(byte-recompile-directory \".\" 0 'force)")))
                    ((require 'elpaca))
                    ((elpaca-generate-autoloads "elpaca" repo)))
              (progn (message "%s" (buffer-string)) (kill-buffer buffer))
            (error "%s" (with-current-buffer buffer (buffer-string))))
        ((error) (warn "%s" err) (delete-directory repo 'recursive))))
    (unless (require 'elpaca-autoloads nil t)
      (require 'elpaca)
      (elpaca-generate-autoloads "elpaca" repo)
      (let ((load-source-file-function nil)) (load "./elpaca-autoloads"))))

  (defun omer/run-python ()
    "Run current Python file."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "py")
      (user-error "Not a Python file"))
    (save-buffer)
    (python-shell-send-buffer)
    (python-shell-switch-to-shell))

  (global-set-key (kbd "C-c p") #'omer/run-python)

  (defun omer/run-c ()
    "Compile and run C code."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "c")
      (user-error "Not a C file"))
    (save-buffer)
    (let ((output (file-name-sans-extension (buffer-file-name))))
      (async-shell-command
       (format "gcc %s -o %s && %s"
               (shell-quote-argument (buffer-file-name))
               (shell-quote-argument output)
               (shell-quote-argument output))
       "*C Output*")
      (switch-to-buffer-other-window "*C Output*")))

  (global-set-key (kbd "C-c c") #'omer/run-c)

  (defun omer/run-rust ()
    "Run `cargo run` for current Rust project."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "rs")
      (user-error "Not a Rust file"))
    (save-buffer)
    (let ((default-directory
           (or (locate-dominating-file default-directory "Cargo.toml")
               (user-error "Not inside a Cargo project"))))
      (let ((buf (get-buffer-create "*Rust Terminal*")))
        (switch-to-buffer-other-window buf)
        (unless (derived-mode-p 'vterm-mode)
          (vterm-mode))
        (vterm-send-string "cargo run")
        (vterm-send-return))))

  (global-set-key (kbd "C-c r") #'omer/run-rust)

  (defun omer/run-lua ()
    "Run current Lua file."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "lua")
      (user-error "Not a Lua file"))
    (save-buffer)
    (async-shell-command
     (format "lua %s" (shell-quote-argument (buffer-file-name)))
     "*Lua Output*")
    (switch-to-buffer-other-window "*Lua Output*"))

  (global-set-key (kbd "C-c l") #'omer/run-lua)

  (defun omer/run-lisp ()
    "Evaluate Common Lisp buffer using SLY."
    (interactive)
    (unless (derived-mode-p 'lisp-mode)
      (user-error "Not a Common Lisp buffer"))
    (sly-eval-buffer))

  (global-set-key (kbd "C-c e") #'omer/run-lisp)

  (defun my/dired-open-file ()
    "Open file at point. Use async external viewer for media/docs, normal for text files."
    (interactive)
    (let* ((file (dired-get-filename nil t))
           (ext (and file (file-name-extension file))))
      (if (and ext (member (downcase ext)
                           '("pdf" "mp4" "mkv" "avi" "mov" "mp3" "flac"
                             "wav" "ogg" "m4a" "webm" "doc" "docx"
                             "xls" "xlsx" "ppt" "pptx")))
          (start-process "dired-open" nil "xdg-open" file)
        (dired-find-file))))

  (with-eval-after-load 'dired
    (define-key dired-mode-map (kbd "RET") 'my/dired-open-file))

  (defun my-open-in-gwenview-rs ()
