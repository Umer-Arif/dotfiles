  (defun omer/run-python ()
    "Run current Python file."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "py")
      (user-error "Not a Python file"))
    (save-buffer)
    (python-shell-send-buffer)
    (python-shell-switch-to-shell))

  (global-set-key (kbd "C-c p") #'omer/run-python)

  (defun omer/run-c ()
    "Compile and run C code."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "c")
      (user-error "Not a C file"))
    (save-buffer)
    (let ((output (file-name-sans-extension (buffer-file-name))))
      (async-shell-command
       (format "gcc %s -o %s && %s"
               (shell-quote-argument (buffer-file-name))
               (shell-quote-argument output)
               (shell-quote-argument output))
       "*C Output*")
      (switch-to-buffer-other-window "*C Output*")))

  (global-set-key (kbd "C-c c") #'omer/run-c)

  (defun omer/run-rust ()
    "Run `cargo run` for current Rust project."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "rs")
      (user-error "Not a Rust file"))
    (save-buffer)
    (let ((default-directory
           (or (locate-dominating-file default-directory "Cargo.toml")
               (user-error "Not inside a Cargo project"))))
      (let ((buf (get-buffer-create "*Rust Terminal*")))
        (switch-to-buffer-other-window buf)
        (unless (derived-mode-p 'vterm-mode)
          (vterm-mode))
        (vterm-send-string "cargo run")
        (vterm-send-return))))

  (global-set-key (kbd "C-c r") #'omer/run-rust)

  (defun omer/run-lua ()
    "Run current Lua file."
    (interactive)
    (unless (string= (file-name-extension (buffer-file-name)) "lua")
      (user-error "Not a Lua file"))
    (save-buffer)
    (async-shell-command
     (format "lua %s" (shell-quote-argument (buffer-file-name)))
     "*Lua Output*")
    (switch-to-buffer-other-window "*Lua Output*"))

  (global-set-key (kbd "C-c l") #'omer/run-lua)

    "Open file at point. Use async external viewer for media/docs, normal for text files."
    (interactive)
    (let* ((file (dired-get-filename nil t))
           (ext (and file (file-name-extension file))))
      (if (and ext (member (downcase ext)
                           '("pdf" "mp4" "mkv" "avi" "mov" "mp3" "flac"
                             "wav" "ogg" "m4a" "webm" "doc" "docx"
                             "xls" "xlsx" "ppt" "pptx")))
          (start-process "dired-open" nil "xdg-open" file)
        (dired-find-file))))

  (defun my/auto-start-eglot-when-path-ready (buffer program-name)
    "Poll the buffer until envrc injects the path containing PROGRAM-NAME, then start Eglot."
    (when (buffer-live-p buffer)
      (with-current-buffer buffer
        (if (executable-find program-name)
            ;; The language server binary is finally found in the PATH! Start Eglot.
            (eglot-ensure)
          ;; Binary not found yet, check again in 0.1 seconds
          (run-with-timer 0.1 nil #'my/auto-start-eglot-when-path-ready buffer program-name)))))

  (defun my/trigger-nix-eglot-sequence ()
    "Initialize envrc and figure out which binary to watch for based on major-mode."
    (envrc-mode 1)

                        ((derived-mode-p 'python-ts-mode) "pyright-langserver")
                        ((derived-mode-p 'typescript-ts-mode) "typescript-language-server")
  					  ((derived-mode-p 'nix-ts-mode) "nil")
                        ((derived-mode-p 'tsx-ts-mode) "typescript-language-server"))))
      (when binary-name
        (my/auto-start-eglot-when-path-ready (current-buffer) binary-name))))
